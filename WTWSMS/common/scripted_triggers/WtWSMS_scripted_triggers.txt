# The Bureaucratic government type is used for the WRE and ERE
bureaucratic_government_trigger = {
	#higher_tier_than = king
	OR = {
		title = e_wre
		title = e_byzantium
		title = e_roman_empire
		title = k_soissons
		title = k_italy
	}	
	NOT = {
		has_title_flag = wre_restored
		has_title_flag = ere_restored
	}
}
# The Sassanid government type is used for the WRE and ERE
sassanid_government_trigger = {
	OR = {
		top_liege = {
			title = e_persia
		}
		title = e_persia
		OR = {
			culture = persian 
			culture = parthian 
		}
	}
}
