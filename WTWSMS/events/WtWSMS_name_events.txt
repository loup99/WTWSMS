#Western Convertion - Flavius Anicius Olybrius
character_event = {
	id = imperial_names.1
	title = EVTNAMEIN1
	desc = EVTDESCIN1
	picture = GFX_evt_family
	only_playable = yes
	
	is_triggered_only = yes

	trigger = {
		AND = {
			character = 1000023
			has_landed_title = e_wre
			has_romance_culture_trigger = yes
			prisoner = no
			ai = no
			NOT = { has_character_flag = change_name_check }
		}
	}
	
	option = {
		name  = EVTWESTO1
		set_name = "Honorius"
		hidden_tooltip = {
			set_character_flag = change_name_check
		}
	}
	option = {
		name  = EVTWESTO2
		set_name = "Valentinian"
		hidden_tooltip = {
			set_character_flag = change_name_check
		}
	}
	option = {
		name  = EVTWESTO3
		set_name = "Honorius-Valentinian"
		hidden_tooltip = {
			set_character_flag = change_name_check
		}
	}
}

#Western Convertion - Octavian-Seleucid connection
character_event = {
	id = imperial_names.7
	title = EVTNAMEIN7
	desc = EVTDESCIN7
	picture = GFX_evt_family
	only_playable = yes
	
	is_triggered_only = yes	


	trigger = {
		AND = {
			character = 1102680
			has_romance_culture_trigger = yes
			has_landed_title = d_carthago_inferior
			ai = no
			prisoner = no
			NOT = { has_character_flag = change_name_check }
		}
	}
	
	option = {
		name  = EVTWESTAUGSEL1
		dynasty = 104246212
		culture = greek
		religion = hellenic_pagan
		add_weak_claim = e_seleucid
		any_child_even_if_dead = { dynasty = 104246212 }
		hidden_tooltip = {
			set_character_flag = change_name_check
		}
	}
	option = {
		name  = EVTWESTAUGSEL2
		hidden_tooltip = {
			set_character_flag = change_name_check
		}
	}
}
