	}
}

# To the Emperor (fires for head of Nicene religion)
character_event = {
	id = HRE.4
	title = EVTNAMEHRE.4
	desc = EVTDESCHRE.4
	
	picture = GFX_evt_great_schism
	
	trigger = {
		year = 950 #Why this year? Temporary anyway.
		religion = nicene 
		controls_religion = yes
		has_landed_title = k_nicene
		NOT = { has_global_flag = great_schism }
	}
	
	option = {
		name = EVTOPTAHRE.4

		set_parent_religion = { religion = arian parent = catholic }
		set_parent_religion = { religion = valentinianism parent = catholic }
		set_parent_religion = { religion = donatism parent = catholic }
		set_parent_religion = { religion = docetic parent = orthodox }
		set_parent_religion = { religion = ebionites parent = orthodox }
		set_parent_religion = { religion = marcionism parent = orthodox }
		set_parent_religion = { religion = binitarian parent = orthodox }
		set_parent_religion = { religion = melchisedechianism parent = orthodox }
		set_parent_religion = { religion = sethian parent = catholic }
		set_parent_religion = { religion = eutychian parent = nestorian } #Nontrinitarian beliefs extremely similar to Nestorianism
		set_parent_religion = { religion = antinomianism parent = nestorian } #Antinomianism is one of the most ancient sects, and most like the Messalians that are also Antinomians, and Messalians are counted as Nestorian heresies
		set_parent_religion = { religion = monatist parent = catholic }
		set_parent_religion = { religion = total_depravity parent = catholic }
		set_parent_religion = { religion = apollinarian parent = orthodox } #Similar to Monothelites ("Jesus is controlled by Skynet", AKA no mind of His own)
		set_parent_religion = { religion = adoptionism parent = miaphysite } #Nontrinitarian, and monarchism is a Miaphysite heresy already, this is a branch of it built on Jesus' baptism
		set_parent_religion = { religion = docetic parent = catholic }
		set_parent_religion = { religion = pneumatomachians parent = orthodox } #Founded by Macedonius, a Bishop of Constantinople
		set_parent_religion = { religion = patripassianism parent = miaphysite } #Monarchism is a Miaphysite heresy already, and this is a form of it
		set_parent_religion = { religion = sabellianism parent = miaphysite } #Nontritarian monarchists again
		set_parent_religion = { religion = psilanthropism parent = orthodox }
		set_parent_religion = { religion = priscillianism parent = catholic } #My Manicheanism claim was a mistake, Priscillian was a Catholic clergyman with extreme Gnostic influences
		
		
		set_global_flag = great_schism
		
		any_playable_ruler = {
			limit = {
				religion = nicene
				controls_religion = no
			}
			narrative_event = { id = HRE.5 }	
		}
	}
	option = {
		name = EVTOPTBHRE.3
	}
}
