# Maintenance events

namespace = misc_event

# Destroy Bagaude
character_event	= {
	id = misc_event.1
	desc = misc_event.1.desc
	hide_window = yes	
	
	ai = yes
	only_rulers = yes
	
	trigger = {
		has_landed_title = d_bagaudae
		ROOT = { 
			NOT = { 
				primary_title = {
					title = d_bagaudae
				}
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	
	option = {
		name = OK
		
		activate_title	= { 
			title = d_bagaudae 
			status = no 
		}
	}
}

# Adds not recognised by Senate modifier to all new independent holders of the WRE or Italy
character_event = { 
	id = misc_event.6
	
	is_triggered_only = yes
	
	hide_window = yes
	
	only_playable = yes
	only_rulers = yes
	only_independent = yes
	
	trigger = {
		OR = {
			has_landed_title = e_wre
			has_landed_title = k_italy
		}
		is_title_active	= d_senate 
		NOT = { has_character_flag = had_senate_recognition_event }
	}
	
	immediate = {
		if = {
			limit = {
				has_landed_title = e_wre
			}
			e_wre = {
				set_title_flag = not_recognised_senate
			}
		}
		if = {
			limit = {
				has_landed_title = k_italy
			}
			k_italy = {
				set_title_flag = not_recognised_senate
			}
		}
	}
}
