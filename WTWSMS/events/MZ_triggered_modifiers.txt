character_event = { # Zoroastrian receives penalty for not being married
	id = 1000094008
	hide_window = yes

	potential = {
		religion_group = zoroastrian_group
	}
	
	trigger = {
		age = 16
		religion_group = zoroastrian_group
		is_married = no
	}
	option = {
		name = "All the singles, put your hands up"
		add_character_modifier = {
			name = zoroastrian_not_married
			duration = -1
		}
	}
}

character_event = { # Zoroastrian loses penalty for not being married
	id = 1000094009
	hide_window = yes

	potential = {
		religion_group = zoroastrian_group
	}
	
	trigger = {

	}
}
