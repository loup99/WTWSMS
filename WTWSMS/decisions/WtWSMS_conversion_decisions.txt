decisions = {

	# Pagans can convert to the Nicene faith if their wife or a concubine is Nicene
	convert_to_spouse_nicene = {
			
		icon_religion = nicene
	
		potential = {
			ai = no
			is_playable = yes
			religion_group = pagan_group
			controls_religion = no
			NOT = { trait = zealous }
			OR = {
				any_spouse = {
					religion = nicene
					hidden_tooltip = {
						is_alive = yes
					}
				}
				any_consort = {
					religion = nicene
					hidden_tooltip = {
						is_alive = yes
					}
				}
			}
		}
		
		effect = {
			prestige = -500
			religion_authority = {
				modifier = ruler_converted_from
			}
			
			if = {
				limit = {
					trait = cynical
				}
				random = {
					chance = 60
					add_trait = sympathy_pagans
				}
			}
			if = {
				limit = {
					NOT = { trait = cynical }
				}
				random = {
					chance = 50
					add_trait = sympathy_pagans
				}
			}		
			
			religion = nicene
			
			hidden_tooltip = {
				religion_authority = {
					modifier = ruler_converted_to
				}
			}
			piety = 200
		}
		
		revoke_allowed = {
			always = no
		}
		
		ai_will_do = {
			factor = 0
		}
	}
